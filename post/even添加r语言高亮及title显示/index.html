even添加r语言高亮及title显示

安装scoop，用scoop下载yarn和node.js

#1 第一步打开powershell（不是cmd）

Set-ExecutionPolicy RemoteSigned -scope CurrentUser

#下载并安装scoop
iwr -useb get.scoop.sh | iex

#安装yarn
scoop install yarn

#安装nodejs
scoop install nodejs

#验证是否安装成功
yarn --version

修改配置文件

修改even\layouts\partials\scripts.html

<!-- 拷贝对r的高亮支持的扩展文件，其他语言类似-->


{{- if .Site.Params.highlightInClient -}}
  <script src="{{ "lib/highlight/highlight.pack.js?v=20171001" | relURL }}"></script>
 <!--  注释里是要拷贝的内容


<script
 charset="UTF-8"
 src="https://cdnjs.cloudflare.com/ajax/libs/highlight.js/9.12.0/languages/r.min.js"></script>


-->
{{- end -}}

修改even\src\css\variables.scss

//在 Code type list.一栏添加对r的映射

language-r: "R",

重新编译

#在站点文件夹下调出命令行窗口修改


$ cd ./themes/even/

# install dependencies
$ yarn install

# build
$ yarn build

发布

#测试是否成功修改配置文件
$ hugo server

测试

x <- 5
y <- 6
